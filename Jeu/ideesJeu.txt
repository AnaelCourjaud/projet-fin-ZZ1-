Gestion Etat Jeux

case Accueil
    afficher ecran Accueil 
    -->Lore1
    nbvague = NB_VAGUE_TOT

case Lore1  
    son Lore1
    afficher ecran Lore1
    --> Lore2

case Lore2 
    son Lore2
    afficher ecran Lore2
    --> Lore3

case Lore3
    son Lore3
    afficher ecran Lore3
    --> Vague

case Vague:
    choix vague attaque --> sortie  = liste attaque
    animation(arriveeattaque)
    calcul total type attaque (cac/tir) FONCTION
    nbriposte = NB_RIPOSTE_VAGUE

case AttenteRiposte:
    affichage(attente)
    si PV = 0 --> case FinduJeu
    choix une riposte --> case riposte
    choix ensemble des defenses pour ce tour (ia = markov, joueur = touche)
        entrée = liste des attaques
        compte nb attaque
        sortie =  liste defense quand taille liste defense = taille listeattaque
    
case Riposte
    nb riposte ++
    calcul degat infligé sur stat total
    calcul ennemi tués (modif liste attaque : entièrement tué = mise à 0/ partiellement tué = modif) --> sortie = liste attaque
    affiche(attaque, defense, resolution) --> ennemi à 0 = tué/ affichage riposte (zone)
    suppression des attaquants tués
    -->retour sur case AttenteRiposte sauf si nbriposte = 0 --> case FindeVague

case FindeVague
    resolution PV pour chauqe monstre restant
    affichage(attaque)
    modif(Markov) (si IA)
    nbvague --
    -->retour sur Vague sauf si vague = 0 --> case FinduJeu

case FinduJeu:
    si vague = 0 : afficherfond(victoire)
    si PV = 0 : afficher fond defaite
